## Usage

```jsx
import { useState } from 'react'
import { Button } from '@aragon/ui'

const App = () => {
  const [clicks, setClicks] = useState(0)
  <Button onClick={() => setClicks(clicks+1)}>{clicks} clicks</Button>
}
```
